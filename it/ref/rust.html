static 定数

一度しか初期化しないが、コンパイル時に初期値が定まらない値を使いたいとき。

fn dst_dir() -> &'static PathBuf {
    static BUILD_DIR: OnceLock<PathBuf> = OnceLock::new();
    BUILD_DIR.get_or_init(|| {
        std::env::current_dir()
            .expect("Failed to get the current directry")
            .join("build")
    })
}
