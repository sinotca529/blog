Segmenter によるトーカナイズ

async function tokenize(text) {
    if (!('Segmenter' in Intl)) {
        alert("このブラウザはSegmenterをサポートしていません。");
        return null;
    }
    const colors = ["#ffcccc", "#ccffcc", "#ccccff","#ccffff", "#ffccff", "#ccffff"];

    const segmenter = new Intl.Segmenter("ja", { granularity: "word" });
    const segments = segmenter.segment(text)[Symbol.iterator]();

    const output = document.createDocumentFragment();
    segments.forEach((segment, idx) => {
        const span = document.createElement("span");
        span.textContent = segment.segment;
        span.style.backgroundColor = colors[idx % colors.length];
        output.appendChild(span);
    });

    return output;
}
