めぐる式二分探索

範囲を [ok, ng)か (ng, ok] で持つ。 終了条件は |ok - ng| = 1。

let mut ok = 0;    // 解が存在する値
let mut ng = 1000; // 解が存在しない値

while (ok - ng).abs() > 1 {
    let mid = (ok + ng) / 2;
    if cond(mid) {
        ok = mid;
    } else {
        ng = mid;
    }
}

return ok;
