基本構文

#!/bin/bash

# if 文
echo "Enter a number:"
read number
if [ $number -gt 10 ]; then
    echo "The number is greater than 10."
elif [ $number -eq 10 ]; then
    echo "The number is exactly 10."
else
    echo "The number is less than 10."
fi

# while 文
counter=1
while [ $counter -le 5 ]; do
    echo "Counter: $counter"
    ((counter++))
done

# for 文
for i in {1..5}; do
    echo "Iteration: $i"
done

# case 文
echo "Enter a letter (a, b, c):"
read letter
case $letter in
    a)
        echo "You selected A.";;
    b)
        echo "You selected B.";;
    c)
        echo "You selected C.";;
    *)
        echo "Invalid selection.";;
esac

# 関数
my_function() {
    echo "This is a function."
    echo "You passed: $1"
}

my_function "Hello, Bash!"

小数の比較

bash は小数の比較ができない。 小数を比較するには bc か awk を使う。
